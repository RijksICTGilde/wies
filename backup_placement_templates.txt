<script>
function selectColleague(name, id) {
  // Scroll to the manual form section
  document.querySelector('form').scrollIntoView({ behavior: 'smooth' });
  
  // Pre-fill the colleague field if it exists
  const colleagueField = document.querySelector('select[name="colleague"]');
  if (colleagueField) {
    // Try to find and select the colleague option
    const options = colleagueField.options;
    for (let i = 0; i < options.length; i++) {
      if (options[i].text.includes(name)) {
        colleagueField.selectedIndex = i;
        break;
      }
    }
    
    // Trigger change event to update any dependent fields
    colleagueField.dispatchEvent(new Event('change'));
  }
  
  // Show a confirmation message
  const messageDiv = document.createElement('div');
  messageDiv.className = 'rvo-notification rvo-notification--success rvo-layout-margin-bottom--md';
  messageDiv.innerHTML = `
    <div class="rvo-notification__icon">
      <span class="utrecht-icon rvo-icon rvo-icon-vinkje rvo-icon--md rvo-icon--wit"></span>
    </div>
    <div class="rvo-notification__content">
      <p class="utrecht-paragraph rvo-paragraph--no-margins">
        ${name} geselecteerd. Pas indien nodig de details hieronder aan.
      </p>
    </div>
  `;
  
  // Insert the message before the manual form
  const form = document.querySelector('form');
  form.parentNode.insertBefore(messageDiv, form);
  
  // Remove the message after 5 seconds
  setTimeout(() => {
    if (messageDiv.parentNode) {
      messageDiv.parentNode.removeChild(messageDiv);
    }
  }, 5000);
}
</script>
